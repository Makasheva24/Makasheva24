- –í–æ–∑—Ä–∞—Å—Ç, —à–∞–≥–∏ ‚Äî **—Ü–µ–ª—ã–µ —á–∏—Å–ª–∞** –±–µ–∑ –¥—Ä–æ–±–Ω—ã—Ö —á–∞—Å—Ç–µ–π, –∏—Ö –ø—Ä–æ—â–µ –ø—Ä–æ—Å—Ç–æ —Ä–∞–Ω–¥–æ–º–Ω–æ –±—Ä–∞—Ç—å.

**–ò—Ç–æ–≥:**

–í—Å—ë –∑–¥–µ—Å—å –ø–æ—Å—Ç—Ä–æ–µ–Ω–æ —Ç–∞–∫, —á—Ç–æ–±—ã **—Å–∏–º—É–ª–∏—Ä–æ–≤–∞—Ç—å —Ä–µ–∞–ª—å–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ**, –∞ –Ω–µ –ø—Ä–æ—Å—Ç–æ "–Ω–∞—Ñ–∏–≥–∞—á–∏—Ç—å —Ä–∞–Ω–¥–æ–º–Ω—ã—Ö —Ü–∏—Ñ—Ä".

```python
import polars as pl
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# 1. –°–æ–∑–¥–∞–Ω–∏–µ –∏ –∑–∞–≥—Ä—É–∑–∫–∞ –Ω–∞–±–æ—Ä–∞ –¥–∞–Ω–Ω—ã—Öüìä

# –£—Å—Ç–∞–Ω–æ–≤–∫–∞ —Å–ª—É—á–∞–π–Ω–æ–≥–æ –∑–µ—Ä–Ω–∞ –¥–ª—è –≤–æ—Å–ø—Ä–æ–∏–∑–≤–æ–¥–∏–º–æ—Å—Ç–∏
np.random.seed(42)

n_samples = 1000

# –ì–µ–Ω–µ—Ä–∞—Ü–∏—è –¥–∞–Ω–Ω—ã—Ö

        "age": np.random.randint(18, 70, size=n_samples), # –≤–æ–∑—Ä–∞—Å—Ç
        "gender": np.random.choice(["Male", "Female"], size=n_samples), # –ø–æ–ª
        "height_cm": np.random.normal(170, 10, size=n_samples).round(1), # —Ä–æ—Å—Ç
        "weight_kg": np.random.normal(70, 15, size=n_samples).round(1), # –≤–µ—Å
        "daily_steps": np.random.randint(1000, 20000, size=n_samples), # —à–∞–≥–∏ –≤ –¥–µ–Ω—å
        "hours_sleep": np.random.normal(7, 1.5, size=n_samples).round(1), # —Å–æ–Ω –≤ —á–∞—Å–∞—Ö
        "activity_level": np.random.choice(["Low", "Medium", "High"], size=n_samples), # —É—Ä–æ–≤–µ–Ω—å –∞–∫—Ç–∏–≤–Ω–æ—Å—Ç–∏
}

# –ü–µ—Ä–µ–≤–æ–¥ –≤ Pandas DataFrame
df_pd = pd.DataFrame(data)

# –ü–µ—Ä–µ–≤–æ–¥ –≤ Polars DataFrame
df_pl = pl.DataFrame(data)

# –°–æ—Ö—Ä–∞–Ω—è–µ–º –≤ CSV —á–µ—Ä–µ–∑ Pandas
df_pd.to_csv("health_data.csv", index=False)

print("‚úÖ –î–∞–Ω–Ω—ã–µ —É—Å–ø–µ—à–Ω–æ —Å–≥–µ–Ω–µ—Ä–∏—Ä–æ–≤–∞–Ω—ã –∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω—ã –≤ 'health_data.csv'.")
print(df_pd.head())  # –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –ø–µ—Ä–≤—ã–µ 5 —Å—Ç—Ä–æ–∫

# 2. –û–±—Ä–∞–±–æ—Ç–∫–∞ –¥–∞–Ω–Ω—ã—Ö (—á–∏—Å—Ç–∫–∞) üîç

# –ö–æ–ø–∏—Ä—É–µ–º DataFrame, —á—Ç–æ–±—ã –Ω–µ –ø–æ—Ä—Ç–∏—Ç—å –æ—Ä–∏–≥–∏–Ω–∞–ª
df_with_nans = df_pd.copy()

for _ in range(n_nan):

    j = np.random.choice(df_with_nans.columns)
    df_with_nans.at[i, j] = np.nan
    
# ‚úÖ –ü—Ä–æ–≤–µ—Ä–∏–º –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–æ–ø—É—Å–∫–æ–≤
print("üîç –ü—Ä–æ–ø—É—Å–∫–∏ –ø–æ —Å—Ç–æ–ª–±—Ü–∞–º:\n", df_with_nans.isna().sum())

# üìå –ó–∞–ø–æ–ª–Ω–µ–Ω–∏–µ:
# - –ß–∏—Å–ª–æ–≤—ã–µ –∫–æ–ª–æ–Ω–∫–∏ ‚Äî –º–µ–¥–∏–∞–Ω–æ–π
# - –ö–∞—Ç–µ–≥–æ—Ä–∏–∞–ª—å–Ω—ã–µ ‚Äî –º–æ–¥–æ–π

for col in df_with_nans.select_dtypes(include=np.number).columns:
    median = df_with_nans[col].median()
    df_with_nans[col].fillna(median, inplace=True)
    
# –ó–∞–ø–æ–ª–Ω–µ–Ω–∏–µ —Å—Ç—Ä–æ–∫–æ–≤—ã—Ö –∫–æ–ª–æ–Ω–æ–∫ –º–æ–¥–æ–π

    df_with_nans[col].fillna(mode, inplace=True)
    
# ‚úÖ –ü—Ä–æ–≤–µ—Ä–∏–º, –≤—Å–µ –ª–∏ –ø—Ä–æ–ø—É—Å–∫–∏ —É—Å—Ç—Ä–∞–Ω–µ–Ω—ã
print("üßπ –û—Å—Ç–∞–ª–æ—Å—å –ø—Ä–æ–ø—É—Å–∫–æ–≤:", df_with_nans.isna().sum().sum()) # –ü–µ—Ä–≤—ã–π .sum() —Å—á–∏—Ç–∞–µ—Ç –∫–æ–ª-–≤–æ NaN –≤ –∫–∞–∂–¥–æ–π –∫–æ–ª–æ–Ω–∫–µ. –í—Ç–æ—Ä–æ–π .sum() ‚Äî —Å–∫–ª–∞–¥—ã–≤–∞–µ—Ç –≤—Å—ë, —á—Ç–æ–±—ã –ø–æ–ª—É—á–∏—Ç—å –æ–±—â–µ–µ —á–∏—Å–ª–æ NaN

# –ß—Ç–æ —Ç–∞–∫–æ–µ include, inplace, fillna?
```
